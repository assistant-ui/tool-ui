A horizontal carousel for displaying products in comparison shopping scenarios. Optimized for surfacing product recommendations in chat.

<ProductListPresetExample preset="keyboards" />

## When to Use Product List

- **Good for:** Product comparisons, shopping recommendations, featured items
- **Not for:** Long product catalogs (use DataTable), single product details (use MediaCard)

## Key Features

<FeatureGrid>
  <Feature icon="ShoppingCart" title="Comparison shopping">

  </Feature>
  <Feature icon="ArrowLeftRight" title="Carousel navigation">

## Usage

```tsx
// Backend tool
import { tool } from "ai";
import { z } from "zod";
import { SerializableProductListSchema } from "@/components/tool-ui/product-list";

const searchProducts = tool({
  description: "Search for products matching criteria",
  inputSchema: z.object({
    query: z.string(),
    maxPrice: z.number().optional(),
  }),
  outputSchema: SerializableProductListSchema,
  async execute({ query, maxPrice }) {
    const products = await fetchProducts(query, maxPrice);
    return {
      id: `product-search-${query}`,
      products: products.map((p) => ({
        id: p.id,
        name: p.name,
        price: `$${p.price.toFixed(2)}`,
        image: p.imageUrl,
        actions: [
          { id: "view", label: "View Details" },
          { id: "cart", label: "Add to Cart", variant: "default" },
        ],
      })),
    };
  },
});

// Frontend with assistant-ui
import { makeAssistantToolUI } from "@assistant-ui/react";
import { ProductList } from "@/components/tool-ui/product-list";

export const SearchProductsUI = makeAssistantToolUI({
  toolName: "searchProducts",
  render: ({ result }) => (
    <ProductList
      {...result}
      onProductClick={(id) => router.push(`/products/${id}`)}
      onProductAction={(productId, actionId) => {
        if (actionId === "cart") {
          addToCart(productId);
        }
      }}
    />
  ),
});
```
