## When to Use Social Post

- **Good for:** Social media posts, activity feeds, external messages
- **Not for:** Dense event tables, editable rich text

## Key Features

- **Platform styling**: X, LinkedIn, Instagram
- **Author details**: Avatar, name, handle, verified badge
- **Engagement metrics**: Likes, replies, reposts, views
- **Timestamp formatting**: Relative or absolute dates
- **Action buttons**: Reply, open, share + custom interactions

## Usage

```tsx
// Backend tool
import { tool } from 'ai';
import { serializableSocialPostSchema } from '@tool-ui/social-post';

const showTweet = tool({
  description: 'Show a tweet as a social post',
  outputSchema: serializableSocialPostSchema,
  async execute({ id }) {
    const tweet = await fetchTweet(id);
    return {
      id: tweet.id,
      platform: 'x',
      author: {
        name: tweet.author.name,
        handle: `@${tweet.author.username}`,
        avatarUrl: tweet.author.avatarUrl,
      },
      createdAt: tweet.createdAt,
      body: tweet.text,
      meta: { source: 'X', url: tweet.url },
      metrics: {
        likes: tweet.likeCount,
        replies: tweet.replyCount,
        reposts: tweet.retweetCount,
      },
    };
  },
});

// Frontend with assistant-ui
import { makeAssistantToolUI } from '@assistant-ui/react';
import { SocialPost } from '@tool-ui/social-post';

export const ShowTweetUI = makeAssistantToolUI({
  toolName: 'showTweet',
  render: ({ result }) => (
    <SocialPost
      {...result}
      actions={[
        { id: 'open', label: 'Open' },
        { id: 'reply', label: 'Reply' },
      ]}
      onAction={(id, post) => {
        if (id === 'open') window.open(post.meta?.url, '_blank');
        if (id === 'reply') openReplyComposer(post.id);
      }}
    />
  ),
});
```
